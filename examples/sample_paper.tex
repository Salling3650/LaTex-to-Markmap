\section{Advanced Topics}

\subsection{Real Analysis}

\begin{definition}
A sequence $(a_n)$ converges to $L$ if for every $\epsilon > 0$, there exists $N$ such that for all $n > N$, we have $|a_n - L| < \epsilon$.
\end{definition}

\begin{lemma}
Every convergent sequence is bounded.
\end{lemma}

\subsection{Complex Analysis}

The Cauchy integral formula states:

\begin{equation}
f(z_0) = \frac{1}{2\pi i} \oint_C \frac{f(z)}{z - z_0} dz
\end{equation}

where $C$ is a simple closed contour containing $z_0$.
